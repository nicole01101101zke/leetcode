1010 Radix (25 åˆ†)

Given a pair of positive integers, for example, 6 and 110, can this equation 6 = 110 be true? The answer is `yes`, if 6 is a decimal number and 110 is a binary number.

Now for any pair of positive integers *N*1 and *N*2, your task is to find the radix of one number while that of the other is given.

### Input Specification:

Each input file contains one test case. Each case occupies a line which contains 4 positive integers:

```
N1 N2 tag radix
```

Here `N1` and `N2` each has no more than 10 digits. A digit is less than its radix and is chosen from the set { 0-9, `a`-`z` } where 0-9 represent the decimal numbers 0-9, and `a`-`z` represent the decimal numbers 10-35. The last number `radix` is the radix of `N1` if `tag` is 1, or of `N2` if `tag` is 2.

### Output Specification:

For each test case, print in one line the radix of the other number so that the equation `N1` = `N2` is true. If the equation is impossible, print `Impossible`. If the solution is not unique, output the smallest possible radix.

### Sample Input 1:

```in
6 110 1 10
```

### Sample Output 1:

```out
2
```

### Sample Input 2:

```in
1 ab 1 2
```

### Sample Output 2:

```out
Impossible
```

```java
import java.util.*;
import java.io.*;

public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        String input = sc.nextLine();
        String[] strs = input.split(" ");
        String N1 = strs[0];
        String N2 = strs[1];
        int tag = Integer.valueOf(strs[2]);
        long r1 = Long.valueOf(strs[3]);
        
        if(tag==2){
            String temp = N1;
            N1 = N2;
            N2 = temp;
        }
        
        int maxdigit = 0;
        for(int i=0;i<N2.length();i++){
            int c;
            if(N2.charAt(i)>='0' && N2.charAt(i)<='9'){
                c = N2.charAt(i)-'0';
            }else{
                c = N2.charAt(i)-'a'+10;
            }
            if(c>maxdigit){
                maxdigit = c;
            }
        }
        
        long left = maxdigit+1;
        long right = 2147483647;
        long value = trans1(N1,r1);
        if(trans1(N2,maxdigit+1)==value){
            System.out.print(maxdigit+1);
            return;
        }
        while(left<=right){
            long mid = (left+right)/2;
            if(trans2(N2,mid,value)==0){
                System.out.print(mid);
                return;
            }
            if(trans2(N2,mid,value)==1){
                right = mid - 1;
            }
            if(trans2(N2,mid,value)==-1){
                left = mid + 1;
            }
        }
        System.out.print("Impossible");
        
        
    }
    
    public static long trans1(String s,long r){
        long sum = 0;
        long t = 1;
        for(int i=s.length()-1;i>=0;i--){
            if(s.charAt(i)>='0' && s.charAt(i)<='9'){
                sum = sum + (s.charAt(i)-'0')*t;
            }else{
                sum = sum + (s.charAt(i)-'a'+10)*t;
            }
            t = t*r;
        }
        return sum;
    }
    public static int trans2(String s,long r,long value){
        long sum = 0;
        long t = 1;
        for(int i=s.length()-1;i>=0;i--){
            if(t>value){
                return 1;
            }
            if(s.charAt(i)>='0' && s.charAt(i)<='9'){
                sum = sum + (s.charAt(i)-'0')*t;
            }else{
                sum = sum + (s.charAt(i)-'a'+10)*t;
            }
            t = t*r;
        }
        if(sum>value){
            return 1;
        }else if(sum==value){
            return 0;
        }else{
            return -1;
        }
    }
    
}
```
